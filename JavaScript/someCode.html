  //иҮӘе·ұеҶҷзҡ„з®ҖеҚ•зҡ„PromiseпјҢиҝҳжІЎеҶҷе®ҢпјҢйҮҢйқўиҝҳжІЎеҶҷcatchпјҢжӯЈеёёејӮжӯҘиғҪе®һзҺ°пјҢдҪҶжҳҜдёҚиғҪй“ҫејҸ+ејӮжӯҘеҶҷеҲ°дәҶresolvePromiseпјҢ
  let FULFILLED = "fulfilled";
  let REJECT = "reject";
  let PENDING = "pending";

  class newPromise {
    #state = PENDING; // #  йҳІжӯўpromiseеҶ…йғЁзҠ¶жҖҒиў«еӨ–йғЁдҝ®ж”№
    #value = null; // з”ЁдәҺи®°еҪ•иҺ·еҸ–еҲ°зҡ„ж•°жҚ®
    #reason = null; // и®°еҪ•еӨұиҙҘзҡ„еҺҹеӣ 
    #onResolveCallbacks = [];
    #onRejectedCallbacks = [];
    constructor(exectuor) {
      // exectuor promiseA+иҢғејҸи§„иҢғж ҮеҮҶи§„е®ҡеҗҚз§°
      const resolve = (value) => {
        if (this.#state === PENDING) {
          // йҳІжӯўзҠ¶жҖҒиў«дҝ®ж”№
          this.#state = FULFILLED;
          this.#value = value;
          this.#onResolveCallbacks.forEach(fn => fn());//ејӮжӯҘзҡ„ж ёеҝғд»Јз ҒпјҢеҪ“жү§иЎҢresolveзҡ„ж—¶еҖҷеҺ»жү§иЎҢжҲҗеҠҹйҳҹеҲ—
          this.#onResolveCallbacks = []; // жё…з©әйҳҹеҲ—
        }
      };
      const reject = (reason) => {
        if (this.#state === PENDING) {
          // йҳІжӯўзҠ¶жҖҒиў«дҝ®ж”№
          this.#state = REJECT;
          this.#reason = reason;
          this.#onRejectedCallbacks.forEach(fn => fn());
          this.#onRejectedCallbacks = []; // жё…з©әйҳҹеҲ—
        }
        throw Error("Error:" + reason);
      };
      try {
        exectuor(resolve, reject);
      } catch (err) {
        reject(err);
      }
    }
    then = function (onFulfilled, onRejected) {
        // еҸӮж•°ж ЎйӘҢ
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : value => value;
        onRejected = typeof onRejected === 'function' ? onRejected : reason => { throw reason; };

      let deepPromise = new newPromise((resolve,reject)=>{ //й“ҫејҸи°ғз”Ёзҡ„ж ёеҝғпјҢpromise.thenйңҖиҰҒиҝ”еӣһдёҖдёӘж–°зҡ„promiseпјҢжүҚиғҪй“ҫејҸжҺҘзқҖ.
        if (this.#state === FULFILLED) {
          setTimeout(()=>{ // жіЁж„ҸпјҡиҝҷйҮҢзҡ„д»Јз ҒйңҖиҰҒжҳҜејӮжӯҘзҡ„еӣ дёәеҰӮжһңdeepPromiseдёӯзҡ„д»Јз ҒжҳҜеҗҢжӯҘзҡ„ resolvePromiseдёӯйңҖиҰҒдј deepPromiseпјҢиҝҳжІЎжңүз”ҹжҲҗе®Ңе°ұдј дјҡжҠҘй”ҷ
            let result = onFulfilled(this.#value); // иҝҷйҮҢиҺ·еҸ–еҲ°еҪ“еүҚ.thenеӣһи°ғзҡ„иҝ”еӣһеҖј
            resolvePromise(result,resolve,reject,deepPromise); // жү§иЎҢдёӢдёҖж¬Ўзҡ„resolve,е№¶дё”еҲӨж–ӯеҸҜиғҪreturn promiseзҡ„жғ…еҶө еҰӮжһңжҳҜpromise йңҖиҰҒжҠҠеҖјдј йҖ’з»ҷpromise
          })
        }
        if (this.#state === REJECT) {
          let rej = onRejected(this.#value);
          reject(rej);
        }
        if(this.#state === PENDING){//ејӮжӯҘзҡ„ж ёеҝғд»Јз ҒпјҢеҪ“и§ҰеҸ‘thenж–№жі•зҡ„ж—¶еҖҷжҠҠжү§иЎҢеҮҪж•°ж”ҫеңЁйҳҹеҲ—дёӯпјҢи§ҰеҸ‘дәҶresolve/rejectзҡ„ж—¶еҖҷеҶҚеҺ»жү§иЎҢ
          this.#onResolveCallbacks.push(()=>{
            onFulfilled(this.#value);
          })
          this.#onRejectedCallbacks.push(()=>{
            onRejected(this.#value);
          })

        }
      })
      return deepPromise;
    };
  }
 /**
 * з”ЁдәҺеҲӨж–ӯreturnзұ»еһӢжҳҜpromiseиҝҳжҳҜжҷ®йҖҡеҖј
 * @param result еҲӨж–ӯзҡ„еҜ№иұЎ
 * @param resolve resultдёәpromiseзҡ„жғ…еҶөиҰҒжү§иЎҢзҡ„resolve
 * @param reject resultдёәpromiseзҡ„жғ…еҶөиҰҒжү§иЎҢзҡ„reject
 * @param deepPromise з”ЁжқҘи§ЈеҶіreturnиҮӘе·ұзҡ„й”ҷиҜҜ
 * @return {number} ж•°з»„зҡ„жҖ»е’Ң
 */
 function resolvePromise(result,resolve,reject,deepPromise){
    if(result === deepPromise){
        reject(new TypeError(`Uncaught TypeError TypeError: Chaining cycle detected for promise #<Promise>
        --- Promise.then ---`))
    // }
    if(result instanceof newPromise){
        // з®ҖеҶҷ(value=>resolve(value), err => reject(err)) зӯүеҗҢдәҺ(resolve,reject)
        // result.then(value=>resolve(value), err => reject(err))
        result.then(resolve,reject);
    } else{
        resolve(result)
    }
    }
 }
